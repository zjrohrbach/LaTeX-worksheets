  \section*{Momentum}

    \begin{align*}
      F \cdot t &=  \Delta p
    \end{align*}

    \begin{align*}
      \Delta p  &= p_f - p_i
    \end{align*}

    \begin{align*}
        \Sigma p_i &= \Sigma p_f
      & p &=mv
    \end{align*}


  \vs
  \hrule
  \section*{Energy}

      
    \begin{align*}
        W   &=  Fd
      & F_G &=  mg
      & P   &=  \frac{W}{t}
    \end{align*}

    \begin{align*}
        KE    &= \frac{1}{2}mv^2 
      & PE    &= mgh
      & KE_i + PE_i + W &= KE_f + PE_f
    \end{align*}

  \vs
  \hrule
  \section*{Simple Harmonic Motion}

      
    \begin{align*}
        T_P &=  2\pi \sqrt{\frac{L}{g}}
      & T_S &=  2\pi \sqrt{\frac{m}{k}}
      & F_S &=  -kd
      & F_G &=  mg
      & v   &=  f\lambda
    \end{align*}

  \vs
  \hrule
  \section*{Light \& Sound}

      
    \begin{align*}
        v   &=  f\lambda
      & f   &= f_s \left( \frac{v\pm v_o}{v\mp v_s} \right)
      & \frac{1}{d_i} + \frac{1}{d_o} &= \frac{1}{f}
      & M &= \frac{h_i}{h_0} = \frac{-d_i}{d_o}
    \end{align*}

    \begin{align*}
      \text{Speeds of Sound: } &

    \end{align*}
