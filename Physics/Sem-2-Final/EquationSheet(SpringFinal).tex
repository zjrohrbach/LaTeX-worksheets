\section*{Momentum}

  \begin{align*}
    F \cdot t &=  \Delta p
  \end{align*}

  \begin{align*}
    \Delta p  &= p_f - p_i
  \end{align*}

  \begin{align*}
      \Sigma p_i &= \Sigma p_f
    & p &=mv
  \end{align*}
