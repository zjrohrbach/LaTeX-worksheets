      \draw[thin, dotted] (-1*\met,1*\met) 
        grid[step=0.2*\met] (4*\met,-1*\met);

        grid[step=\met] (4*\met,-1*\met);

      \node[mass] at (0,0) {\bf 1};
      \node[mass] at (1*\met,0) {\bf 2};
      \node[mass, minimum size=1cm] at (3*\met,0) {\bf 3};

      \draw[|<->|] (1*\met,-1*\met) 
        ++(0,-0.25) -- ++ (\met,0) 
        node[midway, fill=white] {1 m};
