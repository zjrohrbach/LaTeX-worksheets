\section*{Momentum}

\begin{align*}

  J &\equiv F\Delta t &

  \Sigma p &= \Sigma p' &
  v_A+v_A' &= v_B+v_B' &
  x_{CM} &= \frac{\Sigma m_i x_i}{m_i}
