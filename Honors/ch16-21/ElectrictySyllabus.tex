  F &= \frac{kq_1 q_2} {r^2} &

  V &\equiv \frac{PE}{q} = -\frac{W}{q}
\end{align*}

\subsection*{Circuit Equations}

\begin{align*}
  I &\equiv \frac{\Delta q}{t} &
  V &= IR &
  P &= IV = I^2 R = \frac{V^2}{R}
\end{align*}


\begin{align*}
  R_{eq} &= R_1 + R_2 + \cdots &
  \frac{1}{R_{eq}} &= \frac{1}{R_1} + \frac{1}{R_2} + \vdots
\end{align*}
