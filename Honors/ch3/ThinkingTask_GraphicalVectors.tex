\noindent
  Consider these vectors:
  %
  \begin{align*}
    \vec{A} &= \SI{30}{m} \text{ @ $30^\circ$ W of S} \\
    \vec{B} &= \SI{10}{m} \text{ @ $60^\circ$ N of W} \\
    \vec{C} &= \SI{20}{m} \text{, due East}
  \end{align*}

  \begin{parts}
    \part 
      Draw each vector roughly to scale.
    \part
      Draw $\vec{R}_1= \vec{A}+\vec{B}$
    \part
      Draw $\vec{R}_2= \vec{A}+\vec{B}+\vec{C}$
    \part
      Draw $\vec{R}_3= \vec{A}-\vec{B}$
    \part
      Draw $\vec{R}_4= \vec{B}-\vec{A}$
    \part
      Draw $\vec{R}_5= \vec{C}-2\vec{B}$
      
  \end{parts}
