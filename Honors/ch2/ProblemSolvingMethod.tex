\section*{Equations}

\begin{center}
  \begin{align*}
    \bar{s} &= \frac{d}{\Delta t} &
    \bar{v} &= \frac{\Delta x}{\Delta t} &
    \Delta x &= x_f - x_0
  \end{align*}

  \vspace{1em}

  $\SI{1}{\meter\per\second}=\SI{3.6}{\kilo\meter\per\hour}$

\section*{Problem Solving Method}

\begin{enumerate}
  \item Picture
  \item Knowns/Unknowns
  \item Check units
  \item Pick an equation (solve algebraically)
  \item Plug \& Chug
  \item Answer with units
\end{enumerate}

\section*{Problems}

\begin{questions}
  \question
    A certain object has a velocity of \SI{25}{\kilo\meter\per\hour}.

    \begin{parts}
      \part How much time will it take to fly \SI{150}{\meter}?

      \part How about \SI{360}{\meter}?
    \end{parts}


\pagebreak

  \question
    An airplane travels 2100 km at a velocity of 720 km/h and then encounters a tailwind that boosts its speed to 990 km/h for the next 2800 km.  

    \begin{parts}
      \part What was the total time of the trip?
      \part what was the airplane's average velocity?
    \end{parts}


\end{questions}
